int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init(); // Configure button pin as EXTI

    while(1)
    {
        // main loop idle
    }
}

// EXTI callback
void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
{
    if(GPIO_Pin == GPIO_PIN_0) // Button pin
    {
        HAL_GPIO_TogglePin(GPIOC, GPIO_PIN_13); // LED toggle
    }
}
