int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_CAN_Init();

    CAN_TxHeaderTypeDef txHeader;
    uint8_t txData[3] = {1,2,3};
    uint32_t mailbox;

    txHeader.StdId = 0x321;
    txHeader.DLC = 3;

    while(1)
    {
        HAL_CAN_AddTxMessage(&hcan, &txHeader, txData, &mailbox);
        HAL_Delay(1000);
    }
}
