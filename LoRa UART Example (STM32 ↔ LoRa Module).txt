#include "stm32f4xx_hal.h"
#include <string.h>

UART_HandleTypeDef huart2;
uint8_t rx_data;

int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_USART2_UART_Init(); // LoRa module UART

    // Enable UART receive interrupt
    HAL_UART_Receive_IT(&huart2, &rx_data, 1);

    while(1)
    {
        // Periodically send message
        char msg[] = "Hello LoRa UART!\r\n";
        HAL_UART_Transmit(&huart2, (uint8_t*)msg, strlen(msg), HAL_MAX_DELAY);
        HAL_Delay(1000);
    }
}

// Callback when data received via LoRa UART
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    if(huart->Instance == USART2)
    {
        // Echo received data back (optional)
        HAL_UART_Transmit(&huart2, &rx_data, 1, HAL_MAX_DELAY);

        // Re-enable interrupt for next byte
        HAL_UART_Receive_IT(&huart2, &rx_data, 1);
    }
}
