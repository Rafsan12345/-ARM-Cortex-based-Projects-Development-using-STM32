int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_I2C1_Init();

    SSD1306_Init();
    SSD1306_Clear();
    SSD1306_GotoXY(10,10);
    SSD1306_Puts("Hello STM32", &Font_11x18, 1);
    SSD1306_UpdateScreen();

    while(1)
    {
        HAL_Delay(1000);
    }
}
