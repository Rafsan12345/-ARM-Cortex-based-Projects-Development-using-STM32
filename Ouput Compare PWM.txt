int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_TIM3_Init();

    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);

    while(1)
    {
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, 500); // 50% duty
        HAL_Delay(1000);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, 250); // 25% duty
        HAL_Delay(1000);
    }
}
