int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_SPI1_Init();

    MAX7219_Init(); // initialize MAX7219
    uint8_t pattern[8] = {0x18,0x3C,0x7E,0xFF,0xFF,0x7E,0x3C,0x18};

    while(1)
    {
        for(int i=0; i<8; i++)
        {
            MAX7219_SetRow(i, pattern[i]);
        }
        HAL_Delay(500);
    }
}
